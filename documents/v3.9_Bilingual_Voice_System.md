### 2. VoiceAlert.cpp 수정

```cpp
// ════════════════════════════════════════════════════════════
//  생성자
// ════════════════════════════════════════════════════════════

VoiceAlert::VoiceAlert() {
    // ... 기존 초기화 ...
    
    currentLanguage = LANGUAGE_KOREAN;  // 기본값
}

// ════════════════════════════════════════════════════════════
//  언어 설정 (신규)
// ════════════════════════════════════════════════════════════

void VoiceAlert::setLanguage(Language lang) {
    currentLanguage = lang;
    
    const char* langName = (lang == LANGUAGE_KOREAN) ? "한국어" : "English";
    Serial.printf("[VoiceAlert] 언어 변경: %s\n", langName);
}

Language VoiceAlert::getLanguage() const {
    return currentLanguage;
}

uint8_t VoiceAlert::getFolderNumber(uint8_t baseFolder) {
    uint8_t offset;
    
    switch (currentLanguage) {
        case LANGUAGE_KOREAN:
            offset = FOLDER_OFFSET_KOREAN;
            break;
            
        case LANGUAGE_ENGLISH:
            offset = FOLDER_OFFSET_ENGLISH;
            break;
            
        default:
            offset = FOLDER_OFFSET_KOREAN;  // 폴백
            break;
    }
    
    return baseFolder + offset;
}

// ════════════════════════════════════════════════════════════
//  재생 함수 (수정)
// ════════════════════════════════════════════════════════════

void VoiceAlert::playSystem(SystemVoice voice) {
    uint8_t folder = getFolderNumber(FOLDER_BASE_SYSTEM);  // 01 또는 11
    play(folder, (uint8_t)voice);
}

void VoiceAlert::playState(StateVoice voice) {
    uint8_t folder = getFolderNumber(FOLDER_BASE_STATE);   // 02 또는 12
    play(folder, (uint8_t)voice);
}

void VoiceAlert::playError(ErrorVoice voice) {
    uint8_t folder = getFolderNumber(FOLDER_BASE_ERROR);   // 03 또는 13
    play(folder, (uint8_t)voice);
}

void VoiceAlert::playMaintenance(MaintenanceVoice voice) {
    uint8_t folder = getFolderNumber(FOLDER_BASE_MAINTENANCE);  // 04 또는 14
    play(folder, (uint8_t)voice);
}

void VoiceAlert::playGuide(GuideVoice voice) {
    uint8_t folder = getFolderNumber(FOLDER_BASE_GUIDE);   // 05 또는 15
    play(folder, (uint8_t)voice);
}
```

// UI_Screen_Settings.cpp

void drawSettingsScreen() {
    // ... 기존 설정 ...
    
    // ──── 언어 설정 추가 ────
    int y = 150;
    lcd.setTextColor(TFT_CYAN);

                        TFT_GREEN : TFT_DARKGREY;
    lcd.fillRoundRect(150, y - 5, 60, 35, 8, krColor);
    lcd.setTextColor(TFT_WHITE);
    lcd.setTextSize(2);
    lcd.setCursor(160, y + 5);
    lcd.print("KR");
    
    // 영어 버튼
    uint32_t enColor = (currentLanguage == LANGUAGE_ENGLISH) ? 
                        TFT_GREEN : TFT_DARKGREY;
    lcd.fillRoundRect(220, y - 5, 60, 35, 8, enColor);
    lcd.setCursor(230, y + 5);
    lcd.print("EN");
}

bool handleSettingsTouch(uint16_t x, uint16_t y) {
    // ... 기존 터치 처리 ...
    
    // ──── 언어 전환 버튼 ────
    // 한국어 버튼 (150, 145, 60x35)
    if (x >= 150 && x <= 210 && y >= 145 && y <= 180) {
        currentLanguage = LANGUAGE_KOREAN;
        
        #ifdef ENABLE_VOICE_ALERTS
        voiceAlert.setLanguage(LANGUAGE_KOREAN);
        voiceAlert.playSystem(VOICE_READY);  // "준비 완료"
        #endif
        
        screenNeedsRedraw = true;
        return true;
    }
    
    // 영어 버튼 (220, 145, 60x35)
    if (x >= 220 && x <= 280 && y >= 145 && y <= 180) {
        currentLanguage = LANGUAGE_ENGLISH;
        
        #ifdef ENABLE_VOICE_ALERTS
        voiceAlert.setLanguage(LANGUAGE_ENGLISH);
        voiceAlert.playSystem(VOICE_READY);  // "System ready"
        #endif
        
        screenNeedsRedraw = true;
        return true;
    }
    
    return false;
}

## 🎙️ 음성 파일 준비 가이드

### 한국어 TTS (Naver Clova 권장)

```python
import requests

def generate_korean_voice(text, filename):
    client_id = "YOUR_CLIENT_ID"
    client_secret = "YOUR_CLIENT_SECRET"
    
    url = "https://naveropenapi.apigw.ntruss.com/tts-premium/v1/tts"
    
    headers = {
        "X-NCP-APIGW-API-KEY-ID": client_id,
        "X-NCP-APIGW-API-KEY": client_secret,
    }
    
    data = {
        "speaker": "nara",  # 여성: nara, 남성: jinho
        "volume": "0",
        "speed": "-1",      # 조금 느리게
        "pitch": "0",
        "format": "mp3",
        "text": text
    }
    
    response = requests.post(url, headers=headers, data=data)
    
    with open(f"01/{filename}", "wb") as f:
        f.write(response.content)
    
    print(f"생성 완료: {filename}")

# 사용 예
generate_korean_voice("진공 제어 시스템을 시작합니다", "001_welcome.mp3")
```

### 영어 TTS (Google Cloud 권장)

```python
from google.cloud import texttospeech

def generate_english_voice(text, filename):
    client = texttospeech.TextToSpeechClient()
    
    synthesis_input = texttospeech.SynthesisInput(text=text)
    
    voice = texttospeech.VoiceSelectionParams(
        language_code="en-US",
        name="en-US-Wavenet-D",  # 남성 (C, D)
        # name="en-US-Wavenet-F",  # 여성 (E, F, G)
    )
    
    audio_config = texttospeech.AudioConfig(
        audio_encoding=texttospeech.AudioEncoding.MP3,
        speaking_rate=0.9,  # 조금 느리게
        pitch=0.0,
    )
    
    response = client.synthesize_speech(
        input=synthesis_input,
        voice=voice,
        audio_config=audio_config
    )
    
    with open(f"11/{filename}", "wb") as out:
        out.write(response.audio_content)
    
    print(f"생성 완료: {filename}")

# 사용 예
generate_english_voice("Vacuum control system starting", "001_welcome.mp3")
```
