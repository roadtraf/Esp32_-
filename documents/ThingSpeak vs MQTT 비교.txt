
🔍 ThingSpeak vs MQTT 비교
ThingSpeak (현재 사용 중)
방식: HTTP REST API
프로토콜: HTTPS
플랫폼: ThingSpeak 웹사이트
앱: ThingSpeak 공식 앱 또는 웹 브라우저

데이터 흐름:
ESP32 → HTTPS → ThingSpeak 서버 → 웹 대시보드
                                 → 모바일 브라우저
장점:
✅ 웹 대시보드 자동 생성
✅ 차트/그래프 자동 시각화
✅ 무료 (제한적)
✅ 설정 쉬움
단점:
❌ 실시간성 낮음 (15초 최소)
❌ ThingSpeak 웹사이트 접속 필요
❌ MQTT 앱 사용 불가
MQTT (별도 시스템)
방식: Publish/Subscribe
프로토콜: MQTT
브로커: Mosquitto, HiveMQ, CloudMQTT 등
앱: MQTT Dashboard, IoT MQTT Panel 등

데이터 흐름:
ESP32 → MQTT 브로커 → MQTT 앱
장점:
✅ 실시간 (지연 < 1초)
✅ 양방향 통신
✅ 모바일 앱 다양
✅ 가볍고 빠름
단점:
❌ 별도 브로커 필요
❌ 대시보드 직접 구성
❌ 데이터 저장 별도 필요
📱 각 시스템별 모바일 접근
1. ThingSpeak (현재)

구현 방법
void loop() {
    readSensors();
    
    // ThingSpeak: 15초마다 (분석용)
    if (millis() - lastThingSpeakUpdate > 15000) {
        cloudManager.uploadExtendedData();
        lastThingSpeakUpdate = millis();
    }
    
    // MQTT: 1초마다 (실시간용)
    if (millis() - lastMQTTUpdate > 1000) {
        mqttClient.publish("vacuum/pressure", String(pressure).c_str());
        mqttClient.publish("vacuum/temperature", String(temperature).c_str());
        mqttClient.publish("vacuum/health", String(healthScore).c_str());
        lastMQTTUpdate = millis();
    }
    
    delay(10);
}

Step 2: ESP32 코드 추가
// platformio.ini
lib_deps = 
    knolleary/PubSubClient@^2.8

// Config.h
#define ENABLE_MQTT
#define MQTT_SERVER     "broker.hivemq.com"
#define MQTT_PORT       1883
#define MQTT_USER       ""
#define MQTT_PASS       ""
#define MQTT_CLIENT_ID  "ESP32_Vacuum_001"

// MQTTManager.cpp
#include <PubSubClient.h>

WiFiClient espClient;
PubSubClient mqttClient(espClient);

void mqttPublish() {
    // 토픽별로 발행
    mqttClient.publish("vacuum/pressure", 
                       String(sensorData.pressure).c_str());
    mqttClient.publish("vacuum/temperature", 
                       String(sensorData.temperature).c_str());
    mqttClient.publish("vacuum/current", 
                       String(sensorData.current).c_str());
    mqttClient.publish("vacuum/health", 
                       String(healthScore).c_str());
    mqttClient.publish("vacuum/state", 
                       getStateName(currentState));
}

void mqttCallback(char* topic, byte* payload, unsigned int length) {
    // 명령 수신 (양방향 제어)
    String message;
    for (unsigned int i = 0; i < length; i++) {
        message += (char)payload[i];
    }
    
    // 제어 명령 처리
    if (strcmp(topic, "vacuum/control") == 0) {
        if (message == "START") {
            changeState(STATE_VACUUM_ON);
        } else if (message == "STOP") {
            changeState(STATE_IDLE);
        }
    }
}
